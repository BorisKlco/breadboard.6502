05 INPUT "TEXT TO PRINT";A$
10 REM addr for LCD is on $1000
15 LET LCD = 4096
20 FOR X=0 TO LEN(A$)-1
30 PRINT MID$(A$,X+1,1);"-";ASC(MID$(A$,X+1,1))
35 POKE LCD+X, ASC(MID$(A$,X+1,1))
40 NEXT X
45 REM Null Terminator
50 POKE LCD+LEN(A$), 0

75 PRINT "READ MEMORY ON ADDR:";LCD
100 FOR Y=0 TO LEN(A$)
110 PRINT PEEK(LCD+Y);"-";CHR$(PEEK(LCD+Y))
120 NEXT Y
